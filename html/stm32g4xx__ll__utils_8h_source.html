282__STATIC_INLINE void LL_InitTick(uint32_t HCLKFrequency, uint32_t Ticks)
283{
284 /* Configure the SysTick to have interrupt in 1ms time base */
285 SysTick->LOAD = (uint32_t)((HCLKFrequency / Ticks) - 1UL); /* set reload register */
286 SysTick->VAL = 0UL; /* Load the SysTick Counter Value */
287 SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk |
288 SysTick_CTRL_ENABLE_Msk; /* Enable the Systick Timer */
289}
